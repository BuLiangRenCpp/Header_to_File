cmake_minimum_required(VERSION 3.28)
project(Htf)

# 땍屢긴좆
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${SOFT_LIB})

set(INC_NAME 
    Function_h
    Stream_h)
set(SRC_NAME 
    "Function" 
    Stream)

# 鎧乞쨌쓺
foreach(n ${SRC_NAME})
    aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR}/src/${n} SRC)    
endforeach()
list(APPEND SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/htf.cpp)

foreach(n ${INC_NAME})
    include_directories(${CMAKE_CURRENT_SOURCE_DIR}/inc/${n})  
endforeach()

include_directories(${SOFT_INC} ${CMAKE_CURRENT_SOURCE_DIR}/inc)

# target
link_directories(${SOFT_LIB})
add_library(Htf STATIC ${SRC})
target_link_libraries(Htf Prompt-pathDeal)

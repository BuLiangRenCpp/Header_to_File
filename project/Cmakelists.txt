cmake_minimum_required(VERSION 3.28)
project(Main)

# **************************** config **************************************
set(CMAKE_CXX_STANDARD 17)

# ****************************** path **********************************
set(PRO ${CMAKE_CURRENT_SOURCE_DIR})        # project
set(ROOT ${PRO}/../)
set(BIN ${PRO}/bin) 
set(INC 
    ${PRO}/include
    ${PRO}/include/Function_h
    ${PRO}/include/Ins_h
    ${PRO}/include/Stream_h)
set(LIB ${PRO}/lib)

set(CODE ${PRO}/code)
set(SUB_DIR 
    ${CODE}/prompt-pathDeal_dll
    ${CODE}/htf_lib
    ${CODE}/instruction_lib)


# ********************************* build type *****************************************
# -->  cmake -D D=..
# default release
if(D)
    set(CMAKE_BUILD_TYPE Debug)
elseif(R)
    set(CMAKE_BUILD_TYPE Release)
endif()

foreach(dir ${SUB_DIR})
    add_subdirectory(${dir})
endforeach()

# ************************* exe *************************************

set(EXECUTABLE_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/bin)
set(LIB_NAME 
    Prompt-pathDeal 
    Ins
    Htf-code)

aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR}/src SRC)
include_directories(${INC})

link_directories(${LIB})
add_executable(htf ${SRC})
target_link_libraries(htf ${LIB_NAME})

# ***************************** install *********************************
set(CMAKE_INSTALL_PREFIX ${ROOT}/software)

install (DIRECTORY ${PRO}/include
    DESTINATION .)
install(DIRECTORY ${BIN}
    DESTINATION .)

cmake_minimum_required(VERSION 3.28)
project(Root)

# 定义变量
set(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
set(SOFT ${ROOT}/software)
set(SOFT_BIN ${SOFT}/bin) 
set(SOFT_INC 
    ${SOFT}/inc
    ${SOFT}/inc/Stream_h
    ${SOFT}/inc/Function_h)
set(SOFT_LIB ${SOFT}/lib)

set(CODE ${ROOT}/code)
set(SUB_DIR 
    ${CODE}/prompt-pathDeal_dll
    ${CODE}/htf_lib
    ${CODE}/instruction_lib
    ${SOFT})

# 设置预定义宏
set(CMAKE_CXX_STANDARD 17)
if(D)
    set(CMAKE_BUILD_TYPE Debug)
elseif(R)
    set(CMAKE_BUILD_TYPE Release)
endif()

foreach(dir ${SUB_DIR})
    add_subdirectory(${dir})
endforeach()
